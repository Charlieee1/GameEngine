#version 400 core

in vec3 position;
in vec2 texturedCoords;

out vec3 colour;
out vec2 pass_texturedCoords;

uniform mat4 transformationMatrix;

void main(void) {

	gl_Position = transformationMatrix * vec4(position, 1.0);
	pass_texturedCoords = texturedCoords;
	colour = vec3(position.x+0.5,0.0,position.y+0.5);
}
